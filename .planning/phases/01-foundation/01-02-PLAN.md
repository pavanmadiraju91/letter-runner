1. Create `src/core/events.js` — lightweight pub/sub event bus:

```js
// src/core/events.js
const listeners = new Map();

export const events = {
  on(event, callback) {
    if (!listeners.has(event)) {
      listeners.set(event, []);
    }
    listeners.get(event).push(callback);
    // Return unsubscribe function for cleanup
    return () => this.off(event, callback);
  },

  off(event, callback) {
    const cbs = listeners.get(event);
    if (cbs) {
      const idx = cbs.indexOf(callback);
      if (idx !== -1) cbs.splice(idx, 1);
    }
  },

  emit(event, data) {
    const cbs = listeners.get(event);
    if (cbs) {
      for (let i = 0; i < cbs.length; i++) {
        cbs[i](data);
      }
    }
  },

  clear() {
    listeners.clear();
  }
};
```

Design decisions:
- Use `Map` not object (avoids prototype pollution, faster for many event types)
- Array iteration with index (not forEach) — zero allocation per emit
- Return unsubscribe function from `on()` for easy cleanup
- No `once()` — add when needed (YAGNI)
- No async — game events are synchronous within a frame

2. Create `src/core/canvas.js` — DPR-aware full-viewport canvas:

```js
// src/core/canvas.js
import { events } from './events.js';

let canvas, ctx;
let width, height; // Logical (CSS) dimensions

export function initCanvas() {
  canvas = document.getElementById('game');
  ctx = canvas.getContext('2d');
  resize();
  window.addEventListener('resize', resize);
  return { canvas, ctx };
}

function resize() {
  const dpr = window.devicePixelRatio || 1;
  width = window.innerWidth;
  height = window.innerHeight;

  // Set physical (backing store) dimensions
  canvas.width = width * dpr;
  canvas.height = height * dpr;

  // Set CSS display dimensions
  canvas.style.width = width + 'px';
  canvas.style.height = height + 'px';

  // Scale context so draw calls use logical coordinates
  ctx.scale(dpr, dpr);

  // Pixel art: disable anti-aliasing (must reapply after resize resets context)
  ctx.imageSmoothingEnabled = false;

  events.emit('CANVAS_RESIZE', { width, height });
}

export function getCtx() { return ctx; }
export function getWidth() { return width; }
export function getHeight() { return height; }
```

Critical implementation notes (from research pitfalls):
- Always read `devicePixelRatio` fresh on resize (user may drag window between monitors)
- Setting `canvas.width` resets ALL context state — must reapply `scale()` and `imageSmoothingEnabled` AFTER
- `ctx.scale(dpr, dpr)` means all subsequent draw calls use CSS pixel coordinates (no manual DPR math)
- Export getters (not raw variables) so consumers always get current values after resize

1. Create `src/core/game-loop.js` — requestAnimationFrame loop with delta-time:

```js
// src/core/game-loop.js
import { GAME } from '../config.js';

let lastTime = 0;
let running = false;
let updateFn = null;
let renderFn = null;

function tick(currentTime) {
  if (!running) return;

  // Delta in seconds, capped to prevent spiral-of-death on tab switch
  const dt = Math.min((currentTime - lastTime) / 1000, GAME.MAX_DT);
  lastTime = currentTime;

  updateFn(dt);
  renderFn();

  requestAnimationFrame(tick);
}

export function startLoop(update, render) {
  updateFn = update;
  renderFn = render;
  running = true;
  lastTime = performance.now();
  requestAnimationFrame(tick);
}

export function stopLoop() {
  running = false;
}
```

Design decisions (from research):
- `dt` in seconds — physics uses `position += speed * dt` where speed is px/sec
- `GAME.MAX_DT` (1/30 = 33ms) prevents objects teleporting after tab switch
- `lastTime = performance.now()` before first rAF — prevents NaN/huge first-frame dt
- Separate `update(dt)` and `render()` callbacks — update uses time, render reads state
- No fixed-timestep accumulator (not needed without networked physics or replays)
